## 1. Introduction

This document defines the **canonical data contract**, **workflows**, **governance state machine**, **audit schema**, and **safety framework** for the **DeepSynaps Handbook Generation System**. It serves as the single source of truth for:

- Data structures used by the handbook service
- End-to-end generation and export workflows
- Role-based access control (RBAC) + entitlement gating
- Clinical safety guardrails and content validation
- Audit trail requirements for compliance and traceability

All downstream implementations (database schemas, API endpoints, UI flows, export pipelines) MUST conform to this contract.

## 2. Handbook Data Contract

### 2.1 TypedDict Schema

```python
from typing import TypedDict, Optional, List


class Handbook(TypedDict, total=False):
    """
    Canonical Handbook data contract.

    Represents a generated clinical or educational handbook covering
    neuromodulation interventions (TMS, tDCS, tACS, tRNS, taVNS, TPS,
    PBM, neurofeedback, and related modalities).

    Multi-tenancy:   clinic_id scopes all handbook data.
    Patient guides:  patient_id is set for individualized handbooks.
    Generic guides:  patient_id is null/None.
    """

    # ── Identity ──────────────────────────────────────────────
    handbook_id:        str           # UUID v4, generated by system
    clinic_id:          str           # Multi-tenancy partition key
    patient_id:         Optional[str] # Null = generic/modality-level guide

    # ── Generation Parameters ─────────────────────────────────
    audience:           str           # "clinician" | "patient" | "staff_sop"
    title:              str           # Human-readable title
    condition:          str           # Target clinical condition
    modality:           str           # See MODALITIES enum below
    device:             str           # Specific device name/model
    intervention_type:  str           # e.g. "rTMS", "HF-rTMS", "tDCS-F3-F4"

    # ── Content ───────────────────────────────────────────────
    sections:           List[HandbookSection]
    evidence_links:     List[EvidenceLink]
    safety_disclaimer:  str           # Auto-appended per audience

    # ── Review Lifecycle ──────────────────────────────────────
    review_state:       str           # See REVIEW_STATES enum
    signed_by:          Optional[str] # User ID of signer
    signed_at:          Optional[str] # ISO-8601 timestamp

    # ── Export ────────────────────────────────────────────────
    export_status:      str           # "none" | "pending" | "completed" | "failed"

    # ── Audit & Provenance ────────────────────────────────────
    audit_log:          List[AuditEvent]
    provenance:         str           # "ai_generated" | "clinician_edited" | "template_based"

    # ── Quality Controls ──────────────────────────────────────
    evidence_threshold: str           # "A" | "B" | "C" | "D" | "all"
    reading_level:      str           # "professional" | "advanced" | "standard" | "simple"
    output_format:      str           # "markdown" | "docx" | "pdf"

    # ── Timestamps ────────────────────────────────────────────
    created_by:         str           # User ID of creator
    created_at:         str           # ISO-8601 timestamp
    updated_at:         str           # ISO-8601 timestamp


# ── Enumerations ──────────────────────────────────────────────
MODALITIES = {
    "TMS", "tDCS", "tACS", "tRNS", "taVNS",
    "TPS", "PBM", "neurofeedback", "ECT",
    "DBS", "VNS", "FUS", "other"
}

    "draft",         # Initial state after generation
    "needs_review",  # Flagged by automated checks or peer review

    "superseded",    # Replaced by a newer version
}

PROVENANCE = {"ai_generated", "clinician_edited", "template_based"}

EVIDENCE_THRESHOLDS = {"A", "B", "C", "D", "all"}

READING_LEVELS = {"professional", "advanced", "standard", "simple"}

OUTPUT_FORMATS = {"markdown", "docx", "pdf"}

EXPORT_STATUSES = {"none", "pending", "completed", "failed"}
```

## 4. EvidenceLink Data Contract

```python
class EvidenceLink(TypedDict):
    """
    Reference to a clinical evidence source linked to handbook content.

    Evidence links are attached at both the handbook level (summary)
    and at the section level (specific supporting citations).
    """

    link_id:         str     # UUID v4
    source:          str     # See EVIDENCE_SOURCES enum
    title:           str     # Publication / document title
    authors:         str     # Author list or "et al." string
    year:            int     # Publication year (>= 1900, <= current+1)
    doi:             Optional[str]  # DOI string if available
    evidence_grade:  str     # "A" | "B" | "C" | "D"
    provenance:      str     # See PROVENANCE_TYPES enum
    relevance_score: float   # 0.0 – 1.0, cosine or LLM-based
    clinical_note:   str     # Contextual note for this handbook


EVIDENCE_SOURCES = {
    "pubmed",       # PubMed/MEDLINE indexed article
    "cochrane",     # Cochrane Library systematic review
    "nice",         # NICE guideline
    "apa",          # American Psychiatric Association guideline
    "fda",          # FDA 510(k), PMA, or label
    "internal_db",  # Clinic's own evidence database
    "textbook",     # Peer-reviewed textbook or chapter
    "registry",     # Clinical trial registry entry
    "other",        # Other credible source
}

PROVENANCE_TYPES = {
    "measured",    # Directly measured primary data
    "inferred",    # Reasonably inferred from related data
    "proxy",       # Proxy measure (e.g., sham-controlled RCT data)
    "simulated",   # Computational / mechanistic simulation
}

EVIDENCE_GRADES = {
    "A": "High-quality RCTs, meta-analyses, systematic reviews",
    "B": "Moderate-quality evidence, limited RCTs, cohort studies",
    "C": "Observational studies, case series, expert opinion",
    "D": "Preclinical, anecdotal, theoretical rationale only",
}
```

```python
    Immutable audit trail entry for every significant action

    event_type:     str                 # See AUDIT_EVENT_TYPES enum
    timestamp:      str                 # ISO-8601 with timezone
    user_id:        str                 # Actor performing the action

    handbook_id:    str                 # Affected handbook
    patient_id:     Optional[str]       # Patient context if applicable
    ip_address:     Optional[str]       # Originating IP (anonymized)
    user_agent:     Optional[str]       # Client identifier (anonymized)
```

### 6.3 Generation Steps Detail

#### Step 1: Parameter Validation
- Validate all required fields against constraints
- Normalize condition and modality strings
- Resolve `device` against approved device registry
- Log: `HANDBOOK_PAGE_OPEN`

#### Step 2: Entitlement Check
```python
def check_generation_entitlement(user: User, clinic_id: str) -> bool:
    """
    Verify user can generate handbooks for the given clinic.
    """
    return (
        user.role in {"clinician", "admin", "supervisor"}
        and "handbook_generate" in user.entitlements
        and user.clinic_id == clinic_id
    )
```
- On failure: Log `HANDBOOK_ACCESS_DENIED`, return HTTP 403
- On success: Log `HANDBOOK_ENTITLEMENT_CHECKED`

#### Step 3: Evidence Lookup
```python
def lookup_evidence(
    modality: str,
    condition: str,
    threshold: str = "B",
    max_age_years: int = 10,
    preferred_sources: List[str] = None,
    top_n: int = 50,
) -> List[EvidenceLink]:
    """
    Query evidence database for relevant literature.
    Filters by grade >= threshold, modality, condition.
    Sorts by relevance_score DESC.
    """
```
- Log: `HANDBOOK_EVIDENCE_LOOKUP` with query params and result count

#### Step 4: Section Generation
- Iterate over required sections for the audience
- For each section:
  1. Build prompt: modality, condition, device, section type, evidence context
  2. Generate content via AI (with temperature=0.3 for determinism)
  3. Run forbidden-content scanner
  4. Attach top-k relevant evidence links
  5. Set `generated_by = "ai"`, `reviewed = False`
- If AI fails for a section, fall back to template
- Log: `HANDBOOK_GENERATED` with params, section count, model version

#### Step 5: Safety & Validation
- Append mandatory disclaimer based on audience
- Scan all sections for FORBIDDEN_PHRASES
- Validate required section completeness
- If forbidden content found → block, flag, set `review_state = "needs_review"`, Log `HANDBOOK_CONTENT_BLOCKED`
- If validation passes → set `review_state = "draft"`

#### Step 6: Persistence
- Save handbook to database
- Create initial audit log entries
- Return `HandbookGenerationResponse`

### 7.3 Export Gate Logic

```python
def can_export_handbook(user: User, handbook: Handbook) -> Tuple[bool, str]:
    """
    Determine whether a user can export a handbook.
    Returns (allowed, reason).
    """
    # Role check
    if user.role not in {"clinician", "admin", "supervisor"}:
        return False, "role"

    # Entitlement check
    if "handbook_generate" not in user.entitlements:
        return False, "entitlement"

    # Tenant isolation
    if user.clinic_id != handbook["clinic_id"]:
        return False, "tenant"

    # State machine check: must be "signed" to export
    if handbook["review_state"] != "signed":
        return False, f"state:{handbook['review_state']}"

    # Patient access check (if patient-scoped)
    if handbook.get("patient_id"):
        if not user.can_access_patient(handbook["patient_id"]):
```

### 8.4 State Machine Implementation

```python
class HandbookStateMachine:
    """
    Encapsulates all valid state transitions and their guards.
    """

    def __init__(self, handbook: Handbook):
        self.handbook = handbook

    def can_transition(self, target_state: str, user: User) -> Tuple[bool, str]:
        """
        Check if a state transition is allowed.
        Returns (allowed, reason).
        """
        current = self.handbook["review_state"]

        # Check transition exists
        transitions = STATE_TRANSITIONS.get(current, {})
        if target_state not in transitions:
            return False, f"No transition from {current} to {target_state}"

        rule = transitions[target_state]

        # Check role
        if user.role not in rule["roles"]:
            return False, f"Role {user.role} cannot transition {current} → {target_state}"

        # Check entitlement (if required)
        if rule["entitlement"] and rule["entitlement"] not in user.entitlements:
            return False, f"Missing entitlement: {rule['entitlement']}"

        # Check guard condition
        guard_fn = getattr(self, f"_guard_{rule['guard']}", lambda: True)
        if not guard_fn():
            return False, f"Guard '{rule['guard']}' failed"

        return True, ""

    def transition(self, target_state: str, user: User) -> Handbook:
        """
        Execute state transition with full validation.
        """
        allowed, reason = self.can_transition(target_state, user)
        if not allowed:
            raise StateTransitionError(reason)

        old_state = self.handbook["review_state"]
        self.handbook["review_state"] = target_state
        self.handbook["updated_at"] = utc_now_iso()

        # Audit logging
        audit = build_audit_event(
            event_type=AuditEventType.HANDBOOK_APPROVED if target_state == "approved"
                      else AuditEventType.HANDBOOK_SIGNED if target_state == "signed"
                      else AuditEventType.HANDBOOK_REVIEW_COMPLETED,
            handbook_id=self.handbook["handbook_id"],
            user_id=user.user_id,
            metadata={
                "old_state": old_state,
                "new_state": target_state,
                "transition_guard": STATE_TRANSITIONS[old_state][target_state]["guard"],
            }
        )
        self.handbook["audit_log"].append(audit)

        return self.handbook

    # ── Guard Implementations ─────────────────────────────────

    def _guard_all_sections_reviewed_and_safe(self) -> bool:
        return all(
            section["reviewed"] for section in self.handbook["sections"]
        ) and not self._has_safety_flags()

    def _guard_safety_flag_triggered(self) -> bool:
        return self._has_safety_flags()

    def _guard_all_flags_resolved(self) -> bool:
        return not self._has_safety_flags()

    def _guard_all_issues_resolved(self) -> bool:
        return not self._has_safety_flags() and all(
            section["reviewed"] for section in self.handbook["sections"]
        )

    def _guard_digital_signature_provided(self) -> bool:
        return bool(self.handbook.get("signed_by"))

    def _guard_export_completed(self) -> bool:
        return self.handbook.get("export_status") == "completed"

    def _guard_revision_requested(self) -> bool:
        # Always allow revision from exported state
        return True

    def _has_safety_flags(self) -> bool:
        return any(
            event["event_type"] == AuditEventType.HANDBOOK_SAFETY_FLAGGED
            and event["result"] == "failure"
            for event in self.handbook.get("audit_log", [])
        )


class StateTransitionError(Exception):
    """Raised when an invalid state transition is attempted."""
    pass
```

```python
class EntitlementChecker:
    """
    Centralized entitlement resolution for all handbook actions.
    """

    def __init__(self, user: User):
        self.user = user

    def resolve(self, action: str, handbook: Handbook = None) -> Tuple[bool, str]:
        """
        Check if user can perform the given action.
        Returns (allowed, reason_or_empty).
        """
        matrix_row = ENTITLEMENT_MATRIX.get(action)
        if not matrix_row:
            return False, f"Unknown action: {action}"

        # Role check
        min_role = matrix_row["min_role"]
        if not self._role_meets(min_role):
            return False, f"Requires role >= {min_role}, got {self.user.role}"

        # Entitlement check
        required_entitlement = matrix_row["entitlement"]
        if required_entitlement and required_entitlement not in self.user.entitlements:
            return False, f"Missing entitlement: {required_entitlement}"

        # State check
        if handbook and matrix_row["required_state"]:
            required = matrix_row["required_state"]
            if isinstance(required, set):
                if handbook["review_state"] not in required:
                    return False, f"Requires state in {required}, got {handbook['review_state']}"
            else:
                if handbook["review_state"] != required:
                    return False, f"Requires state={required}, got {handbook['review_state']}"

        # Tenant check
        if handbook and self.user.clinic_id != handbook["clinic_id"]:
            return False, "clinic_id mismatch"

        return True, ""

    def _role_meets(self, min_role: str) -> bool:
        """Check if user's role meets or exceeds the minimum."""
        hierarchy = {
            "viewer": 0, "staff": 1, "clinician": 2,
            "supervisor": 3, "admin": 4,
        }
        return hierarchy.get(self.user.role, -1) >= hierarchy.get(min_role, 999)
```

```python
def scan_forbidden_content(content: str) -> List[Dict[str, Any]]:
    """
    Scan content for forbidden phrases.
    Returns list of violations with phrase, position, and severity.
    """
    violations = []
    content_lower = content.lower()

    for phrase in FORBIDDEN_PHRASES:
        idx = content_lower.find(phrase)
        if idx >= 0:
            violations.append({
                "phrase": phrase,
                "position": idx,
                "context": content[max(0, idx-30):idx+len(phrase)+30],
                "severity": "critical",
                "action": "block_and_flag",
            })

    return violations


def apply_safety_rules(handbook: Handbook) -> Tuple[bool, List[Dict]]:
    """
    Apply all safety rules to a handbook.
    Returns (passed, violations).
    """
    all_violations = []

    for section in handbook["sections"]:
        violations = scan_forbidden_content(section["content"])
        for v in violations:
            v["section_id"] = section["section_id"]
            v["section_title"] = section["title"]
        all_violations.extend(violations)

    # Check required sections
    required = REQUIRED_SECTIONS.get(handbook["audience"], [])
    present_sections = {s["title"].lower().replace(" ", "_") for s in handbook["sections"]}
    # Note: section title matching uses normalized form
    # In practice, the system should map section_id to canonical names

    # Check disclaimer presence
    if not handbook.get("safety_disclaimer"):
        all_violations.append({
            "type": "missing_disclaimer",
            "severity": "critical",
            "action": "block",
        })

    passed = not any(v.get("severity") == "critical" for v in all_violations)
    return passed, all_violations
```

## 12. Validation Rules

### 12.1 Input Validation

```python
from typing import List, Tuple
import re


class HandbookValidator:
    """
    Centralized validation for handbook data contracts.
    """

    def validate_handbook(self, h: Handbook) -> Tuple[bool, List[str]]:
        """Full validation of a handbook record. Returns (valid, errors)."""
        errors = []

        # Required fields
        required = ["handbook_id", "clinic_id", "audience", "title",
                    "condition", "modality", "device", "intervention_type",
                    "sections", "review_state", "created_by", "created_at"]
        for field in required:
            if field not in h or h[field] is None:
                errors.append(f"Missing required field: {field}")

        # Enum validations
        if h.get("audience") not in AUDIENCES:
            errors.append(f"Invalid audience: {h.get('audience')}")
        if h.get("modality") not in MODALITIES:
            errors.append(f"Invalid modality: {h.get('modality')}")
        if h.get("review_state") not in REVIEW_STATES:
            errors.append(f"Invalid review_state: {h.get('review_state')}")
        if h.get("provenance") and h["provenance"] not in PROVENANCE:
            errors.append(f"Invalid provenance: {h['provenance']}")
        if h.get("evidence_threshold") and h["evidence_threshold"] not in EVIDENCE_THRESHOLDS:
            errors.append(f"Invalid evidence_threshold: {h['evidence_threshold']}")
        if h.get("reading_level") and h["reading_level"] not in READING_LEVELS:
            errors.append(f"Invalid reading_level: {h['reading_level']}")
        if h.get("output_format") and h["output_format"] not in OUTPUT_FORMATS:
            errors.append(f"Invalid output_format: {h['output_format']}")
        if h.get("export_status") and h["export_status"] not in EXPORT_STATUSES:
            errors.append(f"Invalid export_status: {h['export_status']}")

        # String length validations
        if h.get("title") and not (3 <= len(h["title"]) <= 200):
            errors.append(f"Title must be 3-200 chars, got {len(h['title'])}")
        if h.get("condition") and not (2 <= len(h["condition"]) <= 200):
            errors.append(f"Condition must be 2-200 chars")

        # Sections
        if not h.get("sections"):
            errors.append("Handbook must have at least one section")
        else:
            for i, section in enumerate(h["sections"]):
                sec_valid, sec_errors = self.validate_section(section)
                if not sec_valid:
                    errors.extend([f"Section {i}: {e}" for e in sec_errors])

        # State consistency
        if h.get("review_state") == "signed":
            if not h.get("signed_by"):
                errors.append("signed_by required when review_state='signed'")
            if not h.get("signed_at"):
                errors.append("signed_at required when review_state='signed'")

        # Safety
        if not h.get("safety_disclaimer"):
            errors.append("safety_disclaimer is required")

        return len(errors) == 0, errors

    def validate_section(self, s: HandbookSection) -> Tuple[bool, List[str]]:
        """Validate a single section."""
        errors = []

        if not s.get("section_id"):
            errors.append("Missing section_id")
        if not s.get("title") or not (1 <= len(s["title"]) <= 200):
            errors.append("Title must be 1-200 characters")
        if not s.get("content"):
            errors.append("Content is required")
        if s.get("generated_by") not in SECTION_GENERATORS:
            errors.append(f"Invalid generated_by: {s.get('generated_by')}")

        return len(errors) == 0, errors

    def validate_evidence_link(self, e: EvidenceLink) -> Tuple[bool, List[str]]:
        """Validate an evidence link record."""
        errors = []

        if not e.get("link_id"):
            errors.append("Missing link_id")
        if e.get("source") not in EVIDENCE_SOURCES:
            errors.append(f"Invalid source: {e.get('source')}")
        if not e.get("title") or not (3 <= len(e["title"]) <= 500):
            errors.append("Title must be 3-500 characters")
        if e.get("year") and not (1900 <= e["year"] <= 2100):
            errors.append("Year must be between 1900 and 2100")
        if e.get("evidence_grade") not in {"A", "B", "C", "D"}:
            errors.append("evidence_grade must be A, B, C, or D")
        if e.get("provenance") not in PROVENANCE_TYPES:
            errors.append(f"Invalid provenance: {e.get('provenance')}")
        score = e.get("relevance_score")
        if score is not None and not (0.0 <= score <= 1.0):
            errors.append("relevance_score must be between 0.0 and 1.0")
        if e.get("doi") and not self._is_valid_doi(e["doi"]):
            errors.append("Invalid DOI format")

        return len(errors) == 0, errors

    @staticmethod
    def _is_valid_doi(doi: str) -> bool:
        """Validate DOI format."""
        pattern = r'^10\.\d{4,}\/.+$'
        return bool(re.match(pattern, doi))
```

### 12.2 Required Sections Validator

```python
def validate_required_sections(handbook: Handbook) -> Tuple[bool, List[str]]:
    """
    Check that all required sections for the audience are present.
    Returns (valid, missing_sections).
    """
    audience = handbook.get("audience")
    required = REQUIRED_SECTIONS.get(audience, [])

    # Map present sections by their canonical name
    present = set()
    for section in handbook.get("sections", []):
        # Normalize section title to canonical form
        canonical = section["title"].lower().replace(" ", "_").replace("-", "_")
        present.add(canonical)
        # Also check against section_id if it follows canonical naming
        present.add(section["section_id"].split("_")[0] if "_" in section["section_id"] else section["section_id"])

    missing = [s for s in required if s not in present]
    return len(missing) == 0, missing
```
